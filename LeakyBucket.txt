
import java.util.Scanner;

public class LeakyBucket {
     public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter bucket capacity: ");
        int bucketCapacity = sc.nextInt();
        System.out.print("Enter output rate (packets per second): ");
        int outputRate = sc.nextInt();
        System.out.print("Enter the number of packets: ");
        int numPackets = sc.nextInt();
        int[] packetarray = new int[numPackets];
        System.out.println("Enter the packet sizes: ");
        for (int i = 0; i < numPackets; i++) {
            packetarray[i] = sc.nextInt();
        }

        int currentBucketSize = 0;

        System.out.println("\nPacket Size\tBucket Size\tSent\tRemaining\tStatus");
        for (int packet : packetarray) {
            if (currentBucketSize + packet <= bucketCapacity) {
                currentBucketSize += packet;
                System.out.println(packet + "\t\t" + currentBucketSize + "\t\t"+Math.min(outputRate,currentBucketSize)+"\t\t"+Math.max(0, currentBucketSize - outputRate)+"\t\tAccepted");
            } else {
                System.out.println(packet + "\t\t" + currentBucketSize+"\t\t"+Math.min(outputRate,currentBucketSize)+"\t\t"+Math.max(0, currentBucketSize - outputRate)+"\t\tDropped");
            }
            currentBucketSize = Math.max(0, currentBucketSize - outputRate);
        }
        sc.close();
    }
}
